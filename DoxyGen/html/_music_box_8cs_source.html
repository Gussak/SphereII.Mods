1 /*
2 * Class: MusicBox
3 * Author: sphereii
4 * Category: Entity
5 * Description:
6 * This script allows the attached block to play music, videos and do animations.
7 *
8 * Music Player:
9 * The music player will read the sounds.xml and play SoundDataNode. The music players are all loot containers; this means you can add an item that contains a "SoundDataNode" that
10 * points to an entry in the sounds.xml. Multiple music items can be added to the loot container, and it will randomly play from them.
11 *
12 * Rather than items in the slots, you can also add the <property name="SoundDataNode" value="mymusic" /> to the block, which will allow it to play without an extra item.
13 *
14 * Animation Player:
15 * When the music box is attached, it will check the Model to find an Animator at the top level. An animator SetBool is called on "IsOn" to turn it on.
16 *
17 * Video Player:
18 * The video player works like the music player, and will play videos listed in the <property name="VideoSource" value="#@modfolder:Resources/MyVideos.unity3d?Squeeze" />.
19 * This assumes that the Squeeze.mp4 video is inside of the MyVideos.unity3d bundle. As in the Music Player, multiple items can be added to the container slots to play from.
20 */

214  public override bool OnBlockActivated(int _indexInBlockActivationCommands, WorldBase _world, int _cIdx, Vector3i _blockPos, BlockValue _blockValue, EntityAlive _player)
215  {
216  #region OnBlockActivated
217 
218  // If there's no transform, no sense on keeping going for this class.
219  BlockEntityData _ebcd = _world.GetChunkFromWorldPos(_blockPos).GetBlockEntity(_blockPos);
220  if (_ebcd == null || _ebcd.transform == null)
221  return false;
222 
223  MusicBoxScript myMusicBoxScript = _ebcd.transform.GetComponent<MusicBoxScript>();
224  if (myMusicBoxScript == null)
225  myMusicBoxScript = _ebcd.transform.gameObject.AddComponent<MusicBoxScript>();
226 
227  bool bRuntimeSwitch = myMusicBoxScript.enabled;
228 
229 
230  // Turn off the music box before we do anything with it.
231  myMusicBoxScript.enabled = false;
232 
233  if (_indexInBlockActivationCommands != 0)
234  {
235  if (_indexInBlockActivationCommands == 1)
236  base.OnBlockActivated(_world, _cIdx, _blockPos, _blockValue, _player);
237 
238  if (_indexInBlockActivationCommands == 2)
239  TakeItemWithTimer(_cIdx, _blockPos, _blockValue, _player);
240  }
241  else
242  {
243  bRuntimeSwitch = !bRuntimeSwitch;
244 
245  // Check if we have an animator and set it
246  myMusicBoxScript.anim = _ebcd.transform.GetComponent<Animator>();
247 
248  // Check if we have a video player as well.
249  myMusicBoxScript.videoPlayer = _ebcd.transform.GetComponent<VideoPlayer>();
250 
251  myMusicBoxScript.myBlockPos = _blockPos;
252  // If the switch is on, then we want to look in the loot container to find a reference to any potential items,
253  // which will over-ride the default audio clip / video clip.
254  if (bRuntimeSwitch)
255  {
256  // We'll try to support getting sounds from multiple sound data nodes, based on all the items in the loot container.
257  List<String> mySounds = new List<String>();
258  List<String> myVideos = new List<String>();
259 
260  TileEntityLootContainer tileLootContainer = (TileEntityLootContainer)_world.GetTileEntity(_cIdx, _blockPos);
261 
262  if (tileLootContainer.items != null)
263  {
264  ItemStack[] array = tileLootContainer.items;
265  for (int i = 0; i < array.Length; i++)
266  {
267  if (array[i].IsEmpty())
268  continue;
269 
270  if (array[i].itemValue.ItemClass.Properties.Values.ContainsKey("OnPlayBuff"))
271  {
272  String Buff = array[i].itemValue.ItemClass.Properties.Values["OnPlayBuff"];
273  _player.Buffs.AddBuff(Buff);
274  }
275 
276  if (array[i].itemValue.ItemClass.Properties.Values.ContainsKey("OnPlayQuest"))
277  {
278  String Quest = array[i].itemValue.ItemClass.Properties.Values["OnPlayQuest"];
279  if (_player is EntityPlayerLocal)
280  {
281  QuestClass myQuest = QuestClass.GetQuest(Quest);
282  if (myQuest != null)
283  (_player as EntityPlayerLocal).QuestJournal.AddQuest(myQuest.CreateQuest());
284  }
285 
286  }
287  // Check for a SoundDataNode for a potential sound clip.
288  if (array[i].itemValue.ItemClass.Properties.Values.ContainsKey("SoundDataNode"))
289  {
290  String strSound = array[i].itemValue.ItemClass.Properties.Values["SoundDataNode"];
291  if (!mySounds.Contains(strSound))
292  mySounds.Add(strSound);
293  }
294  // Check for a video Source for a video clip. If we find it, load the asset and add it to the music box script.
295  if (array[i].itemValue.ItemClass.Properties.Values.ContainsKey("VideoSource"))
296  {
297  // Check if the video source is an asset bundle, and if so, load it directly into the video clip on
298  String strVideo = array[i].itemValue.ItemClass.Properties.Values["VideoSource"];
299  if (strVideo.IndexOf('#') == 0 && strVideo.IndexOf('?') > 0)
300  {
301  if (!myVideos.Contains(strVideo))
302  {
303  myVideos.Add(strVideo);
304  }
305  }
306  }
307  }
308  }
309 
310  // Initialize the data with our defaults.
311  myMusicBoxScript.strAudioSource = this.strAudioSource;
312  myMusicBoxScript.strSoundSource = this.strSoundSource;
313  myMusicBoxScript.strVideoSource = this.strVideoSource;
314  myMusicBoxScript.myEntity = _player;
315 
316  // List of Videos and Sound clips.
317  myMusicBoxScript.VideoGroups = myVideos;
318  myMusicBoxScript.SoundGroups = mySounds;
319 
320  myMusicBoxScript.myVideoClip = null;
321  myMusicBoxScript.enabled = bRuntimeSwitch;
322 
323  }
324  }
325 
326 
327  return false;
328  #endregion
329  }
